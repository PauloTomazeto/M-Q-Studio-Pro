1. GENERATIONSTEP.TSX - CONVERSÃO AUTOMÁTICA DE URLS
================================================================================

Linhas Modificadas: 1-5 (import) + 145-189 (lógica)

ANTES:
------
const image_input = [mainImageUrl];

  image_input.push(mirrorImageUrl);
}

DEPOIS:
-------
let proxyMainUrl = mainImageUrl;
let proxyMirrorUrl = mirrorImageUrl;

try {
  if (mainImageUrl.includes('firebasestorage')) {
    proxyMainUrl = getProxyUrl(mainImageUrl);  // Converte para /api/storage/download/path
  }
} catch (err) {
  console.warn('Fallback para URL original:', err);
}

try {
  if (mirrorImageUrl && mirrorImageUrl.includes('firebasestorage')) {
    proxyMirrorUrl = getProxyUrl(mirrorImageUrl);
  }
} catch (err) {
  console.warn('Fallback para URL original:', err);
}

const image_input = [proxyMainUrl];  // Agora com proxy
if (proxyMirrorUrl) {
  image_input.push(proxyMirrorUrl);
}

IMPACTO:
✓ URLs Firebase são convertidas automaticamente
✓ Se conversão falhar, usa URL original
✓ Suporte a espelhos (mirror images)
✓ Zero impacto se URL não for Firebase
✓ CORS resolvido no backend (proxy)

async diagnoseImage(imageInput: string, sessionId: string): Promise<ScanResult> {
  let formattedImage = imageInput;
  let imageSource = 'unknown';

  // Detecção automática de tipo
  if (imageInput.startsWith('data:image')) {
    imageSource = 'base64_direct';
  } else if (imageInput.startsWith('http://') || imageInput.startsWith('https://')) {
    imageSource = 'signed_url';  // Signed URL ou Proxy URL
  } else if (imageInput.startsWith('blob:')) {
    imageSource = 'blob_url';
  } else if (imageInput.includes('firebasestorage')) {
    imageSource = 'firebase_url';
  } else {
    formattedImage = `data:image/jpeg;base64,${imageInput}`;
    imageSource = 'base64_raw';
  }

  // Envia para API
  // ...

  // Log da fonte
  const scanDocData = { ..., imageSource, createdAt: ... };
  // Salva com rastreamento
}
